cmake_minimum_required(VERSION 3.10)

Project(GraphEngine)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_LINK_WARNING_AS_ERROR ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)


file (GLOB_RECURSE LIB_HEADERS
  Engine/include/*.hpp
)

file (GLOB_RECURSE LIB_SOURCE_FILES
    Engine/src/*.cpp
)


file (GLOB_RECURSE EXECUTABLE_HEADERS
  include/*.hpp
)

file (GLOB_RECURSE EXECUTABLE_FILES
    src/*.cpp
)

add_custom_target(copy_assets ALL
        COMMAND ${CMAKE_COMMAND} -E copy_directory
        "../assets"
        "${CMAKE_BINARY_DIR}/assets"
        COMMENT "Copying assets directory")


#Library and executable definitions
add_library(EngineLib STATIC
  ${LIB_SOURCE_FILES}
  ${LIB_HEADERS}
)

add_executable(EngineExecutable
  ${EXECUTABLE_FILES}
  ${EXECUTABLEHEADERS}
)

# CPM - CMake Package Manager
set(CPM_PATH "cmake/CPM.cmake")
message(STATUS "Using CPM.cmake from ${CPM_PATH}")
include(${CPM_PATH})


# CPM package definitions
set(BUILD_SHARED_LIBS OFF)
CPMAddPackage(
        NAME raylib
        GITHUB_REPOSITORY raysan5/raylib
        GIT_TAG  5.5
)

#Package linkage
target_link_libraries(EngineLib
    PUBLIC raylib
)

include_directories(
  EngineLib
  PUBLIC
  "${CMAKE_CURRENT_SOURCE_DIR}/Engine/include"
)

include_directories(
  EngineExecutable
  PUBLIC
  "${CMAKE_CURRENT_SOURCE_DIR}/include"
)


target_link_libraries(EngineExecutable
    PRIVATE EngineLib
)

add_dependencies(EngineExecutable copy_assets)
